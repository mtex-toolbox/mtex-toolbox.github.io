change reference frame of an ODF

Syntax

{% highlight matlab %} SO3F = transformReferenceFrame(SO3F,cs) {% endhighlight %}

Input

odf SO3Fun
cs new crystalSymmetry

Output

odf SO3Fun

Example

{% highlight matlab %} cs1 = crystalSymmetry('121',[1 3 2]); cs2 = crystalSymmetry('112',[2 1 3]); {% endhighlight %} {% highlight matlab %} ori = orientation.rand(cs1) odf1 = SO3FunHarmonic(unimodalODF(ori)) %odf1 = BinghamODF([1 0 0 0],ori) %odf1 = SO3FunHandle(@(ori) odf1.eval(ori),odf1.CS) odf2 = transformReferenceFrame(odf1,cs2) {% endhighlight %} {% highlight plaintext %} ori = orientation (121 → y↑→x) Bunge Euler angles in degree phi1 Phi phi2 156.958 161.468 197.878 odf1 = SO3FunHarmonic (121 → y↑→x) bandwidth: 25 weight: 1 odf2 = SO3FunHarmonic (112 → y↑→x) bandwidth: 25 weight: 1 {% endhighlight %} {% highlight matlab %} plotPDF(odf1,Miller({1,0,0},{0,0,1},{0,1,0},cs1)) nextAxis(2,1) plotPDF(odf2,Miller({0,1,0},{1,0,0},{0,0,1},cs2)) {% endhighlight %}
