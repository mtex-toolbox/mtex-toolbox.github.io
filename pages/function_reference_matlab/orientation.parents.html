variants of an orientation relationship

Syntax

{% highlight matlab %} ori_parents = ori_child * inv(mori.parents) {% endhighlight %}

Input

mori child to parent orientation relationship
ori_child child orientation

Output

ori_parents all possible parent orientation

Example

parent symmetry

{% highlight matlab %} cs_fcc = crystalSymmetry('m-3m', [3.6599 3.6599 3.6599], 'mineral', 'Iron fcc'); {% endhighlight %}

child symmetry

{% highlight matlab %} cs_bcc = crystalSymmetry('m-3m', [2.866 2.866 2.866], 'mineral', 'Iron bcc') {% endhighlight %} {% highlight plaintext %} cs_bcc = crystalSymmetry mineral : Iron bcc symmetry: m-3m elements: 48 a, b, c : 2.9, 2.9, 2.9 {% endhighlight %}

define a bcc child orientation

{% highlight matlab %} ori_bcc = orientation.goss(cs_bcc) {% endhighlight %} {% highlight plaintext %} ori_bcc = orientation (Iron bcc → y↑→x) Bunge Euler angles in degree phi1 Phi phi2 0 45 0 {% endhighlight %}

define Nishiyama Wassermann fcc to bcc orientation relation ship

{% highlight matlab %} NW = orientation.NishiyamaWassermann (cs_fcc,cs_bcc) {% endhighlight %} {% highlight plaintext %} NW = misorientation (Iron fcc → Iron bcc) (111) || (011) [1-10] || [-100] {% endhighlight %}

compute a fcc parent orientation related to the bcc child orientation

{% highlight matlab %} ori_fcc = ori_bcc * NW {% endhighlight %} {% highlight plaintext %} ori_fcc = orientation (Iron fcc → y↑→x) Bunge Euler angles in degree phi1 Phi phi2 180 54.7356 45 {% endhighlight %}
